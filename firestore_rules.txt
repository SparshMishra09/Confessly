rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anonymous users to read and write confessions
    match /confessions/{document} {
      allow read, write: if request.auth != null && request.auth.token.firebase.sign_in_provider == 'anonymous';
    }
    
    // Allow anonymous users to read and write their own user data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Deny all other requests
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
